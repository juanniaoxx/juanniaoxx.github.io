    class Solution {
public:
    typedef vector<vector<int>> vvi;
    bool possibleToStamp(vector<vector<int>>& grid, int stampHeight, int stampWidth) {
        int n = grid.size();
        int m = grid[0].size();
        vii sum(n + 1, vector<int>(m + 1, 0));
        vii diff(n + 2, vector<int>(m + 2, 0));

        // è®¡ç®—å‰ç¼€å’Œ
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= m; ++j) {
                sum[i][j] = grid[i - 1][j - 1] + sum[i - 1][j] + sum[i][j - 1] - sum[i - 1][j - 1];
            }
        }

        // æ£€æŸ¥æ¯ä¸ªå¯èƒ½æ”¾ç½®é‚®ç¥¨çš„åŒºåŸŸ
        for (int a = 1; a + stampHeight <= n + 1; a++) {
            for (int b = 1; b + stampWidth<= m + 1; b++) {
                // æ³¨æ„å‡1
                int c = a + stampHeight - 1;
                int d = b + stampWidth - 1;
                if (sumRegion(a, b, c, d, sum) == 0) {
                    // æ›´æ–°å·®åˆ†æ•°ç»„
                    diff[a][b] += 1;
                    diff[a][d + 1] -= 1;
                    diff[c + 1][b] -= 1;
                    diff[c + 1][d + 1] += 1;
                }
            }
        }

        // è®¡ç®—å·®åˆ†æ•°ç»„çš„å‰ç¼€å’Œ
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= m; ++j) {
                diff[i][j] += diff[i - 1][j] + diff[i][j - 1] - diff[i - 1][j - 1];
                // æ³¨æ„åç§»é‡
                if (grid[i - 1][j - 1] == 0 && diff[i][j] == 0) {
                    return false;
                }
            }
        }

        return true;
    }

private:
    int sumRegion(int a,int b,int c,int d, vvi &g) {
        return g[c][d] - g[a - 1][d] - g[c][b - 1] + g[a - 1][b - 1];
    }
};

    int fieldOfGreatestBlessing(vector<vector<int>>& forceField) {
        // é¿å…å‡ºç°0.5
        vector<long long> xs, ys;
        for (auto &f : forceField) {
            long long x = f[0], y = f[1], side = f[2];
            xs.push_back(2 * x - side); // å·¦è¾¹ç•Œ
            xs.push_back(2 * x + side); // å³è¾¹ç•Œ
            ys.push_back(2 * y - side); // ä¸Šè¾¹ç•Œ
            ys.push_back(2 * y + side); // ä¸‹è¾¹ç•Œ
        }
        // æ’åºæ•°ç»„å¹¶å»é‡
        sort(xs.begin(), xs.end());
        // uniqueå°†ç›¸é‚»é‡å¤å…ƒç´ ç§»åŠ¨åˆ°æœ«å°¾å¹¶è¿”å›æ–°ç»“å°¾çš„è¿­ä»£å™¨
        // erase(it, end)åˆ é™¤it->endçš„å…ƒç´ 
        xs.erase(unique(xs.begin(), xs.end()), xs.end());
        sort(ys.begin(), ys.end());
        ys.erase(unique(ys.begin(), ys.end()), ys.end());

        // ç¦»æ•£åŒ–
        int n = xs.size(), m = ys.size(), diff[n + 2][m + 2];
        memset(diff, 0, sizeof diff);
        for (auto &f : forceField) {
            long long x = f[0], y = f[1], side = f[2];
            // +1åˆ«å¿˜äº†ï¼Œdiffæ˜¯ä»1å¼€å§‹çš„!
            int a = lower_bound(xs.begin(), xs.end(), 2 * x - side) - xs.begin() + 1; // å·¦è¾¹ç•Œçš„ç›¸å¯¹åºå·
            int c = lower_bound(xs.begin(), xs.end(), 2 * x + side) - xs.begin() + 1; // å³è¾¹ç•Œçš„ç›¸å¯¹åºå·
            int b = lower_bound(ys.begin(), ys.end(), 2 * y - side) - ys.begin() + 1; // ä¸Šè¾¹ç•Œçš„ç›¸å¯¹åºå·
            int d = lower_bound(ys.begin(), ys.end(), 2 * y + side) - ys.begin() + 1; // ä¸‹è¾¹ç•Œçš„ç›¸å¯¹åºå·
            // äºŒç»´å·®åˆ†
            diff[a][b] += 1;
            diff[c + 1][d + 1] += 1;
            diff[c + 1][b] -= 1;
            diff[a][d + 1] -= 1;
        }

        // å‰ç¼€å’Œè¿˜åŸå·®åˆ†æ•°ç»„çš„ç»“æœï¼Œå¹¶è®¡ç®—æœ€å¤§å€¼
        int ans = 0;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                diff[i][j] += diff[i - 1][j] + diff[i][j - 1] - diff[i - 1][j - 1];
                ans = max(ans, diff[i][j]);
            }
        }

        return ans;
    }
